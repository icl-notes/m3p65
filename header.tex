% Macros

\newcommand{\lecture}[3]{
  \marginpar{
    Lecture #1 \\
    #2
    #3
  }
}

\newcommand{\function}[5][]{
  \ifx &#1&
    \begin{array}{rcl}
      #2 & \longrightarrow & #3 \\
      #4 & \longmapsto     & #5
    \end{array}
  \else
    \fullfunction{#1}{#2}{#3}{#4}{#5}
  \fi
}

\newcommand{\correspondence}[2]{
  \cbr{
    \begin{array}{c}
      #1
    \end{array}
  }
  \qquad
  \leftrightsquigarrow
  \qquad
  \cbr{
    \begin{array}{c}
      #2
    \end{array}
  }
}
